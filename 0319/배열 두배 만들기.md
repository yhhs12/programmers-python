def double(numbers):
double_list = [] // numbers의 배열을 2배하여 저장할 리스트 선언한다.
for i in numbers: //nubers에서 하나씩 i를 꺼내와서
double_list.append(i\*2) //그 i를 2배하여 double_list에 추가시킨다.
return double_list //double_list를 리턴한다.
